Creating synthetic data with Python using Faker

April 4, 2024

The basics#

from faker import Faker

fake = Faker()

fake.name()
fake.address()

So in order to create as much data as needed, its as simple as dropping the faker instance into a loop.

for _ in range(10):
  print(fake.name())

Creating a custom provider#

class EmployeeProvider(BaseProvider):
    def role(self, r):
        roles = r # an array of roles, i.e., manager, employee, etc
        return random.choices(roles, weights=[.1, .5, .3, .1])

    def contract(self, t):
        types = t #an array of contract types
        contract_type = random.choices(types, weights=[.6, .3, .1])[0]
        start_date = fake.date_between(start_date=datetime(2020, 1, 1))
        end_date = start_date + relativedelta(months=18) if contract_type == 'fixed-term' else False
        data = {
            'contract_type': contract_type,
            'start_date': start_date,
            'end_date': end_date,
        }
        return data

    def probation(self):
        prob_statuses = [True, False]
        prob_status =  random.choices(prob_statuses, weights=[.05, 0.95])[0]
        start_date = fake.date_between(start_date=datetime(2020, 1, 1)) if prob_status else False
        end_date = start_date + relativedelta(months=3) if prob_status else False
        data = {
            'prob_status': prob_status,
            'prob_start': start_date,
            'prob_end': end_date,
        }
        return data

    def company(self, c):
        companies = c # an array of commpanies
        company_id = random.choices(companies, weights=[.1, .3, .6])[0]
        return company_id

    def vote(self, cands):
        candidates = cands # an array of candidates for an election
        vote = random.choices(candidates, weights=(.02, .03, .05, .2, .06, .07, .08, .1, .15, .09, .13, .01, .03))[0]
        return vote

    def preference(self, cands):
        candidates = cands # an array of candidates for an election
        preference = random.choices(candidates, weights=(.01, .02, .02, .1, .02, .02, .01, .1, .01, .01, .01, .01, .66))[0]
        return preference
